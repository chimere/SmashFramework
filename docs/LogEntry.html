        public function get formattedMessage():String
        {
            var deep:String = "";
            for (var i:int = 0; i < depth; i++)
                deep += "   ";
            
            var reporter:String = "";
            if (reporter)
                reporter = reporter + ": ";
            
            var method:String = "";
            if (method != null && method != "")
                method = method + " - ";
            
            return deep + reporter + method + message;
        }
