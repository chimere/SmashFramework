        public static function codeToString(value:int):String
        {
            var tm:Dictionary = staticTypeMap;
            for (var name:String in tm)
            {
                if (staticTypeMap[name.toUpperCase()].keyCode == value)
                    return name.toUpperCase();
            }
            
            return null;
        }
